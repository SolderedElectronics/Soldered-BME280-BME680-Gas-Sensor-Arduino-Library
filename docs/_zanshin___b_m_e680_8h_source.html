210  if (_I2CAddress) // Using I2C if address is non-zero
211  { //

213  Wire.write(addr); // Send register address to read
214  Wire.endTransmission(); // Close transmission
215  Wire.requestFrom(_I2CAddress, sizeof(T)); // Request 1 byte of data
216  structSize = Wire.available(); // Use the actual number of bytes
217  for (uint8_t i = 0; i < structSize; i++)
218  *bytePtr++ = Wire.read(); // loop for each byte to be read
219  } //
220  else //
221  { //
222  if (_sck == 0) // if sck is zero then hardware SPI
223  { //
224  SPI.beginTransaction(
225  SPISettings(SPI_HERZ, MSBFIRST, SPI_MODE0)); // Start the SPI transaction
226  digitalWrite(_cs, LOW); // Tell BME680 to listen up
227  SPI.transfer(addr | 0x80); // bit 7 is high, so read a byte
228  for (uint8_t i = 0; i < structSize; i++)
229  *bytePtr++ = SPI.transfer(0); // loop for each byte to be read
230  digitalWrite(_cs, HIGH); // Tell BME680 to stop listening
231  SPI.endTransaction(); // End the transaction
232  } else { // otherwise we are using soft SPI
233  int8_t i, j; // Loop variables
234  uint8_t reply; // return byte for soft SPI transfer
235  digitalWrite(_cs, LOW); // Tell BME680 to listen up
236  for (j = 7; j >= 0; j--) // First send the address byte
237  {
238  digitalWrite(_sck, LOW); // set the clock signal
239  digitalWrite(_mosi, ((addr) | 0x80) & (1 << j)); // set the MOSI pin state
240  digitalWrite(_sck, HIGH); // reset the clock signal
241  } // of for-next each bit
242  for (i = 0; i < structSize; i++) // Loop for each byte to read
243  { //
244  reply = 0; // reset our return byte
245  for (j = 7; j >= 0; j--) // Now read the data at that byte
246  { //
247  reply <<= 1; // shift buffer one bit left
248  digitalWrite(_sck, LOW); // set and reset the clock signal
249  digitalWrite(_sck, HIGH); // pin to get the next MISO bit
250  if (digitalRead(_miso)) reply |= 1; // read the MISO bit, add to reply
251  } // of for-next each bit

253  } // of for-next each byte to be read
